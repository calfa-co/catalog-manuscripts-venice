Մարկոսի աւետարանին վերջր կայ գրչէն - մեծագոյն երկաթագրով և կէսն ալ բոլորգիր - այսպէս. Ի Թուականութեան Հայոց։ ՉԼն գրեցաւ տիեզերաքարոզ և գերահռշակ աւետաբեր սուրբ աւետարանս, ձեռամբ անարհեստ և անարժանի՝ Յոհաննէս իրիցու. որ ապաւինեա ի ներողութիւնն աստուծոյ և ի բարերարութիւն նորին, իշխեցի մերձենալ ի հրակէզս և ի բոցաճաճանճ(չ) փայլուածս, որ պատմէ զանքըննիս, զխորս խորոցն աստուծոյ, որում փառք ի յաւիտեանս ամէն,։ 2. Խորաններէն յառաջ (Թղ. 3ա) կայ բնագրէն փոքր ինչ տարբեր, այն է միջակ բոլորգրով, գրչութեան ընդարձակագոյն յիշատակարանը (սկիզբէն պակասաւոր իբր թուղթ մի՝ այսպէս... Իբրև իժի և քարբի խցել զականջ. սըն ոչ ուրեք լուել զբանն տեառն, թէ ամենայն որ հարցի ի վէմս յայս փշրեսցի և յորոյ վերա անկցի հոսեսցէ զնա, որ և հոսեցաւ ի մանջ փոշիս և տուաւ մրրկի մեծի քանզի յանձնէ ել դահիճ անձինն զմարմինս բազկացն կերեալ, ըստ ձայնի սրբոյ հայրապետին։ և այս իբրև այլուստ ուստեք ներգործեցաւ է տեառնէ, զի մի եշխեսցուք համարձակիլ յասել եթէ ամբծութիւն մեր։ և հպարտութիւն ի գլուխ ելանիցէ։ այլ թէ այցելուն աստուած այց առար։ Եւ արդ, բազում զգուշութեամբ զանձինս հոգասցուք, զի մի նախատինք դրացեաց լիցուք, Առ մեզ բազում բան կատարեցաւ և այդ ձեզ յիշումն փոքր ի շատէ։ Ի յայսմ ժամանակի քահանայ ոմն Երեմիաս իմաստուն վաճառական և ակն առատ տեսանող՝ ի մտի եդել զամենայն ինչս իւր վաճառել և սկիզբն արարել յառաջ զստացաւ զտիեերալուր քարոզս և զաստուածաբարբառ մատեանս զհոգիախաղաց տառս։ զհնչումս աւետեաց երկնաւորաց և երկրաւորաց. զյոյսս ամենայն աշխարհի, ևս առաւե, մեծաւ յուսով ստացող սորա Երեմիաս ստացաւ զսա յիշատակ իւր և ծնողաց իւրոց և ազգականաց. և ոչ այսքան այլ և սա շինեաց վկայարան և կատարեաց ի յայսմ ամի և բազում ընչիւք ծախաւորեցաւ ի նմա, մինչև արդ գոյիւ չափ զկեանս իւր զոր ունէր զամենայն, յիշելով զբան սորին սուրբ աւետարանիս եթէ մի հոգայցէք զվաղիւն, քանզի այսու յայտնլ սէր կատարման, եթէ սիրէք զիս զպատուիրանս իմ պահեսջիք։ Եւ արդ աղաչեմ զամենայն աստուածազան ընթերցողսդ և զլսողսդ, զի զայս ամենայն յանձն առ վասն կարի փափագանաց ի պէտ աղաւթից ձերոց յիշելոյ արժանի առնէք ի սուրբ և մաքրափայլ յամբիծ աղաւթս ձեր, զԵրեմիա քահանայ։ և զհայր իւր զՏիրատուք բարեսէր և ողորմած ի պէտս և բարի հասակաւ ծերացեալ և զմայր իւր զՄայրտիկին զփոխեալն ի քրիստոս, և զմիւս սնուցանող մայլ սորա զՏիրանցն որ աշխատեցաւ ի վերա սորա մինչև ի հասակ մեծութեան, և ապա երկու։ սեալ յարդար դատաւորէն զի թողեալ զբնա կութիւն աշխարհիս և զսէր։ տարագրեցաւ է տար աշխարհ բազում ճգնութիւն յանձն առեա վասն փրկութեան անձինն։ Ընդ նմին յիշեսջիք և զկենակից սորա զԹագուհի և զծնաւղսն իւր զՅովաննէս քահանայ, և զմայրն իւր զՄարեմ յիշեսջիք աղաչեմ դարձեալ զԵրեմիաս քահանայ և զեղբարսն իւր՝ զՍարգիս և զԱստուրն, և զքոյլ սոցին զՄարգրիտ, և զԳրիգոր երէց. և արդ քանզի աստուած հասարակաց է պարգևատու և ձրի ի շնորհս, և դուք մի ձանձրանայք այ ախորժելով մտադիրեցէք և ցաւակցաբար խնդրեցէք. յիշեցէք և զՅովաննէս որ ի մէջ եր։ կոցունց շնորհեալ է կնիք լուսաւորութեան սուրբ աւազանին, և զկենակիցն զՏիկնացն, և զքուրն իւր զԹութերն որ յայսմ ամի փոխեցաւ և սուգ մեծ թողել ծնողացն, որոց ողորմեսցի տէր աստուած առ հասարակ ամենեցուն ամէն։ Քանզի և սոքա ի հալալ արդեանց իւրեանց ընծայեցին ի յիշատակութիւն սուրբ աւետարանիս ի փրկութիւն հոգոց իւրեանց։ Եւս առաւել Յովաննէս փահանայի սասոցանո դի և վարդապետի սորա հաւատարիմ և ստացեա, ունի զզարդ կատարմանն զեղբայրսիրութիւն առ ամենիսեան, և եղև սմա առիթ ի հոգևորն և ի մարմնաւորս. որոյ հատուցանէ սմա տէր աստուած վարձս բարիս, և սա կազմեաց զսուրի ատետարանս շնորհելով միայն ի ձիրս։ Եւ զԸնձէր նկարիչ սորա յիշեսջիք աղաչեմ։ Եւ զիս զանարհեստ և զանարժարժան գրողս զՅոհաննէս երէց որ առաւել կարաւտեմ ձերոց աղաւթից. յիշելոյ արժանի արարէք զի։ և զծնողսն իմ, զհայրն իմ Վարդապետ անուն և զմայրն իմ զՄամատիկին, և զերեսին եղբարսն իմ և զքորսն. և ինքն որ տուիչն է ամենայն բարեաց յիսուս քրիստոս առ հասարակ ողորմեսցի յիշողացդ և յիշելեացս և նմա փառյաւիտեանս ամէն։ Գրեցաւ սուրբ աւետարանս ի գաւառիս Կարնս, որ կոչի Թէաւդեպաւլիս, ի Կան գեղ, ըն։   հովանեաւ սրբոյ զաւրավարիս (Սարգսի) և սրբոյն Գրիգորի և սրբոյն Սիոնի և սրբոյն Գէորգա,։ 3. Թղ. 216ա. Կայ բնագրի գրչութեամբ։ փառք ամենաւրհնեալ մարդասիրին աստուծոյ հաւր և որդոյ և սուրբ հոգւոյն այժմ և։ յաւետեանս յաւիտենից որ արժանի արար զիմ անարժանութիւնս հասանել ի վերջ գծիս. Զըստացոզ աւետարանիս՝ զԵրեմիա քահանայ յիշեսջիլ ի սուրբ և ի մաքրափայլ յաղաւթս ձեր. (հոս տոդ մը եղծուած է)։ Ես տեր ավագո2) ստացայ զսուրբ աւետարանս յարդար վաստակոց իմոց ի վայելումն ինձ և զաւակաց իմոց, և յիշատակ ծնաւղաց իմոց, և ամենայն նախնեաց, հաւրն իմոյ Մանկտաւագ(գ) մերչերոյն, և մաւրն իմոյ Մինասթներին, և զուգակցին իմոյ Մամախաթունին, և զաւակաց իմոց Քարիմին և Նեփեմատին և Քիմիլին։ Արդ աղաչեմ զամենեսեան որք ընթեռնուք կամ աւրինակէք, յիշեսջիք, ի մշտապաշտաւն յաղաւթս ձեր. և աստուած զձեզ յիշեսցէ ամէն,։ Այս և նախընթաց, այն է 2. ըդ յիշատակարանին մէջ յիշատակուած նախկին ստացողաց, այսպէս և գրչին անուններուն ձևոց և այլ պարադայից մէջ տարաձայնութիւն կը տեսնուի. վասն զի հոս Երեմիայ քահանային հայրը՝ Մանկտաւագ կոչուած է և մայրը Մինասթեր։ մինչդեռ նախընթացին մէջ հայրը Տիրատուր է և մայրը Մայրտիկին. հետևաբար այս յիշատակարաններէ սվերջնոյն մէջ ձեռք խառնուած է, ինչպէս սեան կէսի տարբեր գիրն իսկ կասկածել կու տայ։ 4. Թղ. 216բ. Կայ, ցանցառ և կէս բու լորգրով, վերջին ստացողէն այսպէս. Իս Աա2 Ղաղարոս որդի ամիր Գրիգորի որ առի սուրբ աւետարանս յիմ հալալ արդեանցս որ ավի յիլ տէրն լիշատակ ինձ և կենակցին իմոյ Ահչի, խաթունին և հարազատ որդոյ իմոյ պըն Թորոսու և ամիր Գրիգորի և Աստուածատուր զաւակին նմաոյ։ և մաւրն իմոյ Մարիանէի առփոխեցեալն։ քրիստոս. և հայրն իմամիր Գրիգորն առփոխեալն ի քրիստոս։ Որք ընթեռնուք զսուրբ աւետարան։ յիշեսջիք ի սուրբ մաքրափայլ յաղաւթս ձեր և զծնաւղսն մեր, և դուք յիշեալ եղիցիք ի միուս անգամ գալըստեանն ամէն։ - Թվին ՊՀԳ։ - և քրիստոսի փառք յաւիտեանս ամէն,։ Այս տեղ նախկին ստացողի յիշատակարանին հետ խառնուած կերև ի անոր յաջորդին յիշատակարանը. վասն զի սեան կէսէն ասդին գիրը կը փոխուի։ 5. Թղ. 1բ. Կայ, նուրբ բոլորգրով, վերջին ստացողի յիշատակարանն, այսպէս. Ե։ Գուլպակիս որդի Աստընվորի (Աստուածատրոյ7) թոռն մադրար Յովանէսին՝ որ գնեցի զսուրբ աւետարանս ի Կսկիմ ի վրաց աղգէ. որ գերեալ էին և տարեալ ի Թվականիս ՋԿԱ. յիշխանութեանն Սպերու պրն։ Կիտեվանին որ տիրեաց ի վրաց աղգէ Արդ ես Գուլպակիս որ գնեցի զսուրբ աւետարանս ի հալալ արդեանց իմոց, յիշատակ ծնաւղացն իմոց Աստընվորին և մաւրն իմ Գոհարին և պապին իմոյ Յովանէսին և մամին իմոյ Սավլուին և ամենայն արեան մերձաւորաց մերոց ի վայե, լումն և ի պահպանութիւն ինձ և եղբարցն իմոց տէր Սարգսին և Տիրատրին, Ամիրին և Չրախին և որդոյ իմոյ ֆիրումին և Դիպալիկին, և եղբաւրորդոյն իմոյ ԱԹէլին և Դովլաթիարին և Սահա Արդ աղաչեմ զձեց ընթերցողք սուրբ աւետարանիս յիշեսջիք ի մաքրափայլ աղաւթ։ ձեր զԳուլպակիս և զկողակիցն իմ զՍուրաթն զտէր Սարգիսն և զկողակիցն իւր զՏանսուխն զՏիրատուրն և զկողակիցն իւր զհանգուցեա Գոհարն, զԱմիրն և զկողակիցն իւր Սուլթանփաշան, զեղբայրն իմ զՉրախն և զքորքն մեր զԱւաքտիկինն, զՍավյուն, զՄիտամն և զԼալխաթունն. և դուք յիշեալ լիջիք ի քրիստոսէ աստուծոյ մերոյ ամէն,։ Ասոր իբրև շարունակութիւն, բայց տարբեր գրչութեամբ անմիջապէս, այսպէս։ Արդ աղաշեմ զձեզ ընդհերջողք սուրբ աւետարանիս յիշեսջիք ի մաքրափայլ յաղաւթս ձեր։ զտէ։ Ղուկաց և զկողակիցն իւր զՀատիւն, զԱբէլ և զկողակիցն իւր զԼուսխաթուն, և զհանկուցեա Շիրինն, զԵարղուղաին և զկողակիցն իւր զՄար։ գարիտն, Սարքիս երէց և զորդին իւր Աստընորն, զքուրն իւր զՏիքարն, զսրդին իւր զԽոճիկն գՄԲխալն, զՂուկաս էրէցն և զորդին իւր զՄերաց՝ և զՂաբարն. և դուք յիշեալ լիջիք ի քրիստո։ սէ աստուծոյ մերոյ ամէն.։
